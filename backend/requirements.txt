# --- Database (SQLAlchemy ORM) ---

alembic         # For database migrations (changing schema over time)

# --- Settings & Environment ---
pydantic-settings # For managing settings and reading from .env files

# --- Security & Authentication ---
python-jose[cryptography] # For creating and verifying JWT tokens
passlib[bcrypt]           # For hashing and verifying passwords
